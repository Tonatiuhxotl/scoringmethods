Welcome to my visualization site. Below is an interactive scatterplot. You can hover over the points, zoom in, and pan around.

library(plotly)

# Simulating a bivariate distribution of latent factors
set.seed(42)
factor1 <- rnorm(500, mean = 0, sd = 1)
factor2 <- 0.7 * factor1 + rnorm(500, mean = 0, sd = 0.7)

# Creating the interactive plotly graph
fig <- plot_ly(
  x = ~factor1, 
  y = ~factor2, 
  type = "scatter", 
  mode = "markers",
  marker = list(size = 6, color = "rgba(70, 130, 180, 0.7)", line = list(color = "white", width = 0.5))
) %>%
  layout(
    title = "Interactive Latent Space",
    xaxis = list(title = "Latent Factor 1 (θ₁)"),
    yaxis = list(title = "Latent Factor 2 (θ₂)")
  )
